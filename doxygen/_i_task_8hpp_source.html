543  void attachMemGetter(std::string name, std::shared_ptr<BaseConnector> connector, MMType type) {
544  std::shared_ptr<ConnectorVector> vector;
545  if (hasMemGetter(name)) {
546  vector = memGetter->find(name)->second;
547  }
548  else {
549  vector = std::shared_ptr<ConnectorVector>(new ConnectorVector());
550  }
551 
552  // If the pipeline id is not the same as the vector size, then the connection has already been made
553  // Each pipeline memory edge is added in the correct order; ie pipeline 0 added first ... pipeline 1 second, etc.
554  // We allow the memory getter to reuse the same memory edge when we have multiple memory releasers for the same memory edge.
555  if (this->pipelineId != vector->size())
556  {
557  return;
558  }
559 
560  vector->push_back(connector);
561 
562  mmTypeMap->insert(std::pair<std::string, MMType>(name, type));
563  memGetter->insert(ConnectorVectorPair(name, vector));
564  DEBUG("Num memory getters " << memGetter->size() << " with " << vector->size() << " connectors");
565  }

577  void attachMemReleaser(std::string name, std::shared_ptr<BaseConnector> connector, MMType type, bool outsideMemManGraph) {
578  std::shared_ptr<ConnectorVector> vector;
579  if (hasMemReleaser(name)) {
580  vector = memReleaser->find(name)->second;
581  }
582  else {
583  vector = std::shared_ptr<ConnectorVector>(new ConnectorVector());
584  }
585 
586  // Errors if you try to add the same memory releaser to the same memory edge.
587  // Each pipeline memory edge is added in the correct order; ie pipeline 0 added first ... pipeline 1 second, etc.
588  // If the same pipieline is added at the incorrect time then the pipeline id will not equal the size of the vector
589  // If the memory edge originates in a graph that this ITask does not belong too, then ignore this check
590  if (this->pipelineId != vector->size() && !outsideMemManGraph)
591  {
592  std::cerr << "Error attempting to add mem releaser to the same named memory edge!" << std::endl;
593  exit(-1);
594  }
595 
596  memReleaserOutsideGraph->insert(std::pair<std::string, bool>(name, outsideMemManGraph));
597 
598  vector->push_back(connector);
599 
600  mmTypeMap->insert(std::pair<std::string, MMType>(name, type));
601  memReleaser->insert(ConnectorVectorPair(name, vector));
602  DEBUG("Num memory releasers " << memReleaser->size() << " with " << vector->size() << " connectors");
603  }

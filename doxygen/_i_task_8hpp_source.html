520  void attachMemGetter(std::string name, std::shared_ptr<BaseConnector> connector, MMType type) {
521  std::shared_ptr<ConnectorVector> vector;
522  if (hasMemGetter(name)) {
523  vector = memGetter->find(name)->second;
524  }
525  else {
526  vector = std::shared_ptr<ConnectorVector>(new ConnectorVector());
527  }
528 
529  // If the pipeline id is the same as the vector size, then the connection has already been made
530  // Each pipeline memory edge is added in the correct order; ie pipeline 0 added first ... pipeline 1 second, etc.
531  // We allow the memory getter to reuse the same memory edge when we have multiple memory releasers for the same memory edge.
532  if (this->pipelineId != vector->size())
533  {
534  return;
535  }
536 
537  vector->push_back(connector);
538 
539  mmTypeMap->insert(std::pair<std::string, MMType>(name, type));
540  memGetter->insert(ConnectorVectorPair(name, vector));
541  DEBUG("Num memory getters " << memGetter->size() << " with " << vector->size() << " connectors");
542  }

552  void attachMemReleaser(std::string name, std::shared_ptr<BaseConnector> connector, MMType type, bool outsideMemManGraph) {
553  std::shared_ptr<ConnectorVector> vector;
554  if (hasMemReleaser(name)) {
555  vector = memReleaser->find(name)->second;
556  }
557  else {
558  vector = std::shared_ptr<ConnectorVector>(new ConnectorVector());
559  }
560 
561  // Errors if you try to add the same memory releaser to the same memory edge.
562  // Each pipeline memory edge is added in the correct order; ie pipeline 0 added first ... pipeline 1 second, etc.
563  // If the same pipieline is added at the incorrect time then the pipeline id will not equal the size of the vector
564  // If the memory edge originates in a graph that this ITask does not belong too, then ignore this check
565  if (this->pipelineId != vector->size() && !outsideMemManGraph)
566  {
567  std::cerr << "Error attempting to add mem releaser to the same named memory edge!" << std::endl;
568  exit(-1);
569  }
570 
571  memReleaserOutsideGraph->insert(std::pair<std::string, bool>(name, outsideMemManGraph));
572 
573  vector->push_back(connector);
574 
575  mmTypeMap->insert(std::pair<std::string, MMType>(name, type));
576  memReleaser->insert(ConnectorVectorPair(name, vector));
577  DEBUG("Num memory releasers " << memReleaser->size() << " with " << vector->size() << " connectors");
578  }

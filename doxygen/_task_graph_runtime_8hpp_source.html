153  void executeRuntime() {
154  if (executed)
155  return;
156 
157  // Initialize graph and setup task graph taskGraphCommunicator
158  this->graph->initialize();
159 
160  std::list<AnyTaskManager *> *vertices = this->graph->getTaskManagers();
161  std::list<AnyTaskManager *> newVertices;
162  DEBUG_VERBOSE("Launching runtime for " << vertices->size() << " vertices");
163  for (AnyTaskManager *task : *vertices) {
164  size_t numThreads = task->getNumThreads();
165 
166  DEBUG_VERBOSE("Spawning " << numThreads << " threads for task " << task->getName());
167 
168  if (numThreads > 0) {
169  std::list<AnyTaskManager *> taskList;
170  std::shared_ptr<std::atomic_size_t>
171  atomicNumThreads = std::shared_ptr<std::atomic_size_t>(new std::atomic_size_t(numThreads));
172  taskList.push_back(task);
173 
174  for (size_t i = 1; i < numThreads; i++) {
175  AnyTaskManager *taskCopy = task->copy(true);
176 
177  // Add communicator to task copy to enable communication
178  taskCopy->setTaskGraphCommunicator(graph->getTaskGraphCommunicator());
179  taskList.push_back(taskCopy);
180  newVertices.push_back(taskCopy);
181  }
182  size_t threadId = 0;
183  for (AnyTaskManager *taskItem : taskList) {
184  TaskManagerThread *runtimeThread = new TaskManagerThread(threadId, taskItem, atomicNumThreads);
185  std::thread *thread = new std::thread(&TaskManagerThread::run, runtimeThread);
186  this->threads.push_back(thread);
187  runtimeThreads.push_back(runtimeThread);
188  threadId++;
189  }
190 
191  } else {
192  std::cerr << task->getName() << "Has no threads specified." << std::endl;
193  }
194  }
195 
196  for (AnyTaskManager *newVertex : newVertices) {
197  graph->addTaskManager(newVertex);
198  }
199 
200  this->executed = true;
201  }
202 
203  private:
204  std::list<std::thread *> threads;
205  AnyTaskGraphConf *graph;

172  void executeTask(std::shared_ptr<T> data) {
173  bool isTerminated = false;
174 
175  output->clear();
176 
177  for (std::shared_ptr<IRule<T, U>> rule : *rules) {
178  {
179  std::unique_lock<std::mutex> lock(rule->getMutex());
180  if (rule->isRuleTerminated(this->pipelineId)) {
181  continue;
182  }
183 
184  DEBUG_VERBOSE("Rule: " << rule->getName() << " consuming data: " << data);
185  std::list<std::shared_ptr<U>> *ruleOut = rule->applyRuleFunction(data, this->pipelineId);
186 
187  if (ruleOut != nullptr) {
188  DEBUG_VERBOSE("Rule: " << rule->getName() << " producing data size: " << ruleOut->size());
189  for (std::shared_ptr<U> o : *ruleOut) {
190  DEBUG_VERBOSE("Rule: " << rule->getName() << " producing data: " << o);
191  output->push_back(o);
192  }
193  }
194 
195  if (rule->isRuleTerminated(this->pipelineId)) {
196  isTerminated = true;
197  }
198  }
199  }
200 
201  if (this->connector != nullptr)
202  this->connector->produceData(output);
203 
204  if (isTerminated) {
205  DEBUG_VERBOSE(this->getName() << " is terminating");
206  this->_isTerminated = true;
207  this->connector->producerFinished();
208  if (this->connector->isInputTerminated()) {
209  this->connector->wakeupConsumer();
210  }

247  void initialize(int pipelineId, int numPipeline, TaskScheduler<T, U> *ownerTask,
248  std::shared_ptr<std::vector<std::shared_ptr<BaseConnector>>> pipelineConnectorList) {
249  this->cudaId = this->cudaIds[pipelineId];
250  this->context = this->contexts[pipelineId];
251 
252  cuCtxSetCurrent(this->context);
253 
254  CUstream stream;
255  cuStreamCreate(&stream, CU_STREAM_DEFAULT);
256 
257  this->stream = stream;
258 
259  CUdevice dev;
260  cuDeviceGet(&dev, this->cudaId);
261 
262  for (int i = 0; i < this->numGpus; i++) {
263  CUcontext ctx = this->contexts[i];
264  if (ctx != this->context) {
265  CUdevice peerDev;
266  cuDeviceGet(&peerDev, this->cudaIds[i]);
267 
268  int canAccessPeer;
269  cuDeviceCanAccessPeer(&canAccessPeer, dev, peerDev);
270 
271  if (canAccessPeer == 0) {
272  this->nonPeerDevIds.push_back(this->cudaIds[i]);
273  this->peerContexts.insert(std::pair<int, CUcontext>(this->cudaIds[i], ctx));
274  }
275  else {
276  cuCtxEnablePeerAccess(ctx, 0);
277  }
278  }
279  }
280 
281  this->initializeCudaGPU(this->context, this->stream, this->cudaId, this->numGpus, pipelineId, numPipeline);
282  }

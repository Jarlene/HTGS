295  void initialize() override final {
296  this->cudaId = this->cudaIds[this->getPipelineId()];
297  this->context = this->contexts[this->getPipelineId()];
298 
299  cuCtxSetCurrent(this->context);
300 
301  CUstream stream;
302  cuStreamCreate(&stream, CU_STREAM_DEFAULT);
303 
304  this->stream = stream;
305 
306  CUdevice dev;
307  cuDeviceGet(&dev, this->cudaId);
308 
309  for (size_t i = 0; i < this->numGpus; i++) {
310  CUcontext ctx = this->contexts[i];
311  if (ctx != this->context) {
312  CUdevice peerDev;
313  cuDeviceGet(&peerDev, this->cudaIds[i]);
314 
315  int canAccessPeer;
316  cuDeviceCanAccessPeer(&canAccessPeer, dev, peerDev);
317 
318  if (canAccessPeer == 0) {
319  this->nonPeerDevIds.push_back(this->cudaIds[i]);
320  this->peerContexts.insert(std::pair<int, CUcontext>(this->cudaIds[i], ctx));
321  } else {
322  cuCtxEnablePeerAccess(ctx, 0);
323  }
324  }
325  }
326 
327  this->initializeCudaGPU();
328  }

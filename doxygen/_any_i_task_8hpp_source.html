565  template<class V>
566  m_data_t<V> getMemory(std::string name, IMemoryReleaseRule *releaseRule, MMType type, size_t nElem) {
567  assert(("Unable to find memory edge 'name' for task", this->memoryEdges->find(name) != this->memoryEdges->end()));
568 
569  auto conn = memoryEdges->find(name)->second;
570  auto connector = std::dynamic_pointer_cast<Connector<MemoryData<V>>>(conn);
571 
572 #ifdef WS_PROFILE
573  sendWSProfileUpdate(StatusCode::WAITING_FOR_MEM);
574 #endif
575 
576  m_data_t<V> memory = connector->consumeData();
577 
578 #ifdef WS_PROFILE
579  sendWSProfileUpdate(StatusCode::EXECUTE);
580 #endif
581 
582  memory->setMemoryReleaseRule(releaseRule);
583 
584  if (memory->getType() != type) {
585  std::cerr
586  << "Error: Incorrect usage of getMemory. Dynamic memory managers use 'getDynamicMemory', Static memory managers use 'getMemory' for task "
587  << this->getName() << " on memory edge " << name << std::endl;
588  exit(-1);
589  }
590 
591  if (type == MMType::Dynamic)
592  memory->memAlloc(nElem);
593 
594  return memory;
595  }

587  template<class V>
588  m_data_t<V> getMemory(std::string name, IMemoryReleaseRule *releaseRule, MMType type, size_t nElem) {
589  assert(("Unable to find memory edge 'name' for task", this->memoryEdges->find(name) != this->memoryEdges->end()));
590 
591  auto conn = memoryEdges->find(name)->second;
592  auto connector = std::dynamic_pointer_cast<Connector<MemoryData<V>>>(conn);
593 
594 #ifdef WS_PROFILE
595  sendWSProfileUpdate(StatusCode::WAITING_FOR_MEM);
596 #endif
597 
598 #ifdef PROFILE
599  auto start = std::chrono::high_resolution_clock::now();
600 #endif
601  m_data_t<V> memory = connector->consumeData();
602 #ifdef PROFILE
603  auto finish = std::chrono::high_resolution_clock::now();
604  this->incMemoryWaitTime(std::chrono::duration_cast<std::chrono::microseconds>(finish - start).count());
605 #endif
606 
607 #ifdef WS_PROFILE
608  sendWSProfileUpdate(StatusCode::EXECUTE);
609 #endif
610 
611 
612 
613  memory->setMemoryReleaseRule(releaseRule);
614 
615  if (memory->getType() != type) {
616  std::cerr
617  << "Error: Incorrect usage of getMemory. Dynamic memory managers use 'getDynamicMemory', Static memory managers use 'getMemory' for task "
618  << this->getName() << " on memory edge " << name << std::endl;
619  exit(-1);
620  }
621 
622  if (type == MMType::Dynamic)
623  memory->memAlloc(nElem);
624 
625  return memory;
626  }

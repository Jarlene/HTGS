544  template<class V>
545  m_data_t<V> getMemory(std::string name, IMemoryReleaseRule *releaseRule, MMType type, size_t nElem) {
546  assert(("Unable to find memory edge 'name' for task", this->memoryEdges->find(name) != this->memoryEdges->end()));
547 
548  auto conn = memoryEdges->find(name)->second;
549  auto connector = std::dynamic_pointer_cast<Connector<MemoryData<V>>>(conn);
550 
551  m_data_t<V> memory = connector->consumeData();
552 
553  memory->setMemoryReleaseRule(releaseRule);
554 
555  if (memory->getType() != type) {
556  std::cerr
557  << "Error: Incorrect usage of getMemory. Dynamic memory managers use 'getDynamicMemory', Static memory managers use 'getMemory' for task "
558  << this->getName() << " on memory edge " << name << std::endl;
559  exit(-1);
560  }
561 
562  if (type == MMType::Dynamic)
563  memory->memAlloc(nElem);
564 
565  return memory;

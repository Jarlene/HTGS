551  template<class V>
552  m_data_t<V> getMemory(std::string name, IMemoryReleaseRule *releaseRule, MMType type, size_t nElem) {
553  assert(("Unable to find memory edge 'name' for task", this->memoryEdges->find(name) != this->memoryEdges->end()));
554 
555  auto conn = memoryEdges->find(name)->second;
556  auto connector = std::dynamic_pointer_cast<Connector<MemoryData<V>>>(conn);
557 
558  m_data_t<V> memory = connector->consumeData();
559 
560  memory->setMemoryReleaseRule(releaseRule);
561 
562  if (memory->getType() != type) {
563  std::cerr
564  << "Error: Incorrect usage of getMemory. Dynamic memory managers use 'getDynamicMemory', Static memory managers use 'getMemory' for task "
565  << this->getName() << " on memory edge " << name << std::endl;
566  exit(-1);
567  }
568 
569  if (type == MMType::Dynamic)
570  memory->memAlloc(nElem);
571 
572  return memory;
573  }

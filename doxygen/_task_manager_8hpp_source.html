136  void executeTask() override {
137 
138  std::shared_ptr<T> data = nullptr;
139 
140  DEBUG_VERBOSE(prefix() << "Running task: " << this->getName());
141 
142  if (this->isStartTask()) {
143  DEBUG_VERBOSE(prefix() << this->getName() << " is a start task");
144  this->setStartTask(false);
145  auto start = std::chrono::high_resolution_clock::now();
146 
147 #ifdef WS_PROFILE
148  this->sendWSProfileUpdate(StatusCode::EXECUTE);
149 #endif
150 
151  this->taskFunction->executeTask(nullptr);
152 
153 #ifdef WS_PROFILE
154  this->sendWSProfileUpdate(StatusCode::WAITING);
155 #endif
156  auto finish = std::chrono::high_resolution_clock::now();
157 
158  this->incTaskComputeTime(std::chrono::duration_cast<std::chrono::microseconds>(finish - start).count());
159  return;
160  } else if (this->taskFunction->canTerminate(this->inputConnector)) {
161 
162  DEBUG(prefix() << this->getName() << " task function is terminated");
163  this->processTaskFunctionTerminated();
164 
165  return;
166  }
167  auto start = std::chrono::high_resolution_clock::now();
168 
169 #ifdef WS_PROFILE
170  this->sendWSProfileUpdate(StatusCode::WAITING);
171 #endif
172  if (this->isPoll())
173  data = this->inputConnector->pollConsumeData(this->getTimeout());
174  else
175  data = this->inputConnector->consumeData();
176 
177  auto finish = std::chrono::high_resolution_clock::now();
178 
179  this->incWaitTime(std::chrono::duration_cast<std::chrono::microseconds>(finish - start).count());
180 
181  DEBUG_VERBOSE(prefix() << this->getName() << " received data: " << data << " from " << inputConnector);
182 
183  if (data != nullptr || this->isPoll()) {
184  start = std::chrono::high_resolution_clock::now();
185 #ifdef WS_PROFILE
186  sendWSProfileUpdate(this->inputConnector.get(), StatusCode::CONSUME_DATA);
187  this->sendWSProfileUpdate(StatusCode::EXECUTE);
188 #endif
189  this->taskFunction->executeTask(data);
190  finish = std::chrono::high_resolution_clock::now();
191 
192  this->incTaskComputeTime(std::chrono::duration_cast<std::chrono::microseconds>(finish - start).count());
193  }

254  void processTaskFunctionTerminated() {
255  // Task is now terminated, so it is no longer alive
256  this->setAlive(false);
257 
258  // Wake up the threads for this task
259  this->getInputConnector()->wakeupConsumer();
260 
261  // If there is a runtime thread, then begin termination
262  if (this->runtimeThread != nullptr) {
263  this->runtimeThread->terminate();
264 
265 #ifdef WS_PROFILE
266  this->sendWSProfileUpdate(StatusCode::SHUTDOWN);
267 #endif
268 
269  // If this is the last thread for this task then close the output
270  if (this->runtimeThread->decrementAndCheckNumThreadsRemaining()) {
271 
272 #ifdef WS_PROFILE
273  // Update output connector, this task is no longer producing for it
274  this->sendWSProfileUpdate(this->getOutputConnector().get(), StatusCode::DECREMENT);
275 #endif
276  if (this->getOutputConnector() != nullptr) {
277  this->getOutputConnector()->producerFinished();
278 
279  if (this->getOutputConnector()->isInputTerminated())
280  this->getOutputConnector()->wakeupConsumer();
281  }
282 
283  auto memManagerConnectorMap = this->getTaskFunction()->getReleaseMemoryEdges();
284 
285  DEBUG(prefix() << " " << this->getName() << " Shutting down " << memManagerConnectorMap->size()
286  << " memory releasers");
287  for (auto nameManagerPair : *memManagerConnectorMap) {
288  DEBUG(prefix() << " " << this->getName() << " Shutting down memory manager: " << nameManagerPair.first);
289 
290 
291  std::shared_ptr<AnyConnector> connector = nameManagerPair.second;
292 #ifdef WS_PROFILE
293  // TODO: This might not be necessary
294  // Update memory manager connector, this task is no longer producing for it
295  this->sendWSProfileUpdate(connector.get(), StatusCode::DECREMENT);
296 #endif
297  connector->producerFinished();
298 
299  if (connector->isInputTerminated())
300  connector->wakeupConsumer();
301  }
302  }
303  } else {
304  if (this->getOutputConnector() != nullptr) {
305  this->getOutputConnector()->producerFinished();
306 
307  if (this->getOutputConnector()->isInputTerminated()) {
308  this->getOutputConnector()->wakeupConsumer();
309  }
310  }
311  }

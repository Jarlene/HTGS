144  void executeTask() override {
145  std::shared_ptr<T> data = nullptr;
146 
147  DEBUG_VERBOSE(prefix() << "Running task: " << this->getName());
148 
149  if (this->isStartTask()) {
150  DEBUG_VERBOSE(prefix() << this->getName() << " is a start task");
151  this->setStartTask(false);
152  auto start = std::chrono::high_resolution_clock::now();
153  this->taskFunction->executeTask(nullptr);
154  auto finish = std::chrono::high_resolution_clock::now();
155 
156  this->incTaskComputeTime(std::chrono::duration_cast<std::chrono::microseconds>(finish - start).count());
157  return;
158  } else if (this->taskFunction->canTerminate(this->inputConnector)) {
159 
160  DEBUG(prefix() << this->getName() << " task function is terminated");
161  this->processTaskFunctionTerminated();
162 
163  return;
164  }
165  auto start = std::chrono::high_resolution_clock::now();
166 
167  if (this->isPoll())
168  data = this->inputConnector->pollConsumeData(this->getTimeout());
169  else
170  data = this->inputConnector->consumeData();
171 
172  auto finish = std::chrono::high_resolution_clock::now();
173 
174  this->incWaitTime(std::chrono::duration_cast<std::chrono::microseconds>(finish - start).count());
175 
176  DEBUG_VERBOSE(prefix() << this->getName() << " received data: " << data << " from " << inputConnector);
177 
178  if (data != nullptr) {
179  start = std::chrono::high_resolution_clock::now();
180  this->taskFunction->executeTask(data);
181  finish = std::chrono::high_resolution_clock::now();
182 
183  this->incTaskComputeTime(std::chrono::duration_cast<std::chrono::microseconds>(finish - start).count());
184  }

310  DEBUG_VERBOSE(prefix() << "Running task: " << this->getName());
311 
312  if (isStartTask) {
313  DEBUG_VERBOSE(prefix() << this->getName() << " is a start task");
314  isStartTask = false;
315  auto start = std::chrono::high_resolution_clock::now();
316  this->taskFunction->executeTask(nullptr);
317  auto finish = std::chrono::high_resolution_clock::now();
318 
319  this->taskComputeTime += std::chrono::duration_cast<std::chrono::microseconds>(finish - start).count();
320  return;
321  } else if (this->taskFunction->isTerminated(this->inputConnector)) {
322 
323 #ifdef PROFILE
324  {
325  std::unique_lock<std::mutex> lock(ioMutex);
326  std::cout << "===================== " << this->getName() << " "<< prefix() << " ===================" << std::endl;
327  std::cout << "COMPUTE TIME: " << taskComputeTime << " us WAIT TIME: " << taskWaitTime << " us" << std::endl;
328 
329  if (this->inputConnector != nullptr) {
330  std::cout << "Input connector: ";
331  this->inputConnector->profileConsume(this->numThreads, true);
332  }
333  if (this->outputConnector != nullptr) {
334  std::cout << "Output connector: ";
335  this->outputConnector->profileProduce(this->numThreads);
336  }
337  this->taskFunction->profileITask();
338  std::cout << "-------------------------- " << this->getName() << " (thread: " << this->threadId << ") -------------------------- " << std::endl << std::endl;
339  }
340 #endif
341 
342 
343  DEBUG(prefix() << this->getName() << " task function is terminated");
344  this->alive = false;
345  this->inputConnector->wakeupConsumer();
346 
347 
348  if (this->runtimeThread != nullptr) {
349  this->runtimeThread->terminate();
350 
351  if (this->runtimeThread->decrementAndCheckNumThreadsRemaining()) {
352  if (this->outputConnector != nullptr) {
353  this->outputConnector->producerFinished();
354 
355  if (this->outputConnector->isInputTerminated())
356  this->outputConnector->wakeupConsumer();
357  }
358

360 
361  DEBUG(prefix() << " " << this->getName() << " Shutting down " << memReleasers->size() <<
362  " memory releasers");
363  for (std::pair<std::string, std::shared_ptr<std::vector<std::shared_ptr<BaseConnector>>> > pair : *memReleasers) {
364 
365 
366  if (this->taskFunction->isMemReleaserOutsideGraph(pair.first))
367  {
368  DEBUG(prefix() << " " << this->getName() << " Shutting down ALL memory releasers : " <<
369  pair.first << " with " << pair.second->size() << " connectors");
370  for (auto connector : *pair.second)
371  {
372  connector->producerFinished();
373 
374 
375  if (connector->isInputTerminated())
376  connector->wakeupConsumer();
377  }
378  }
379  else {
380  DEBUG(prefix() << " " << this->getName() << " Shutting down memory releaser : " <<
381  pair.first << " with " << pair.second->size() << " connectors");
382  std::shared_ptr<BaseConnector> connector = pair.second->at((unsigned long) this->pipelineId);
383  connector->producerFinished();
384 
385 
386  if (connector->isInputTerminated())
387  connector->wakeupConsumer();
388  }
389  }
390 
391  }
392  }
393  else {
394  if (this->outputConnector != nullptr) {
395  this->outputConnector->producerFinished();
396 
397  if (this->outputConnector->isInputTerminated()) {
398  this->outputConnector->wakeupConsumer();
399  }
400  }
401  }
402  return;
403  }
404  auto start = std::chrono::high_resolution_clock::now();
405 
406  if (poll)
407  data = this->inputConnector->pollConsumeData(this->timeout);
408  else
409  data = this->inputConnector->consumeData();
410 
411  auto finish = std::chrono::high_resolution_clock::now();
412 
413  this->taskWaitTime += std::chrono::duration_cast<std::chrono::microseconds>(finish - start).count();
414 
415  DEBUG_VERBOSE(prefix() << this->getName() << " received data: " << data << " from " << inputConnector);
416 
417  if (data != nullptr) {
418  auto start = std::chrono::high_resolution_clock::now();
419  this->taskFunction->executeTask(data);
420  auto finish = std::chrono::high_resolution_clock::now();
421 
422  this->taskComputeTime += std::chrono::duration_cast<std::chrono::microseconds>(finish - start).count();
423  }
